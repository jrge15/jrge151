public class sumamatrices {
    public static int sumamatriz(int X[][],int N, int M){
        int S=0;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                S+=X[i][j];
            }
            
        }
        return S;
    }
    
    public static void llenaM(int N,int M,int X[][]){
        Scanner LEE=new Scanner(System.in);
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                System.out.println("Ingrese elemento"+i+","+j);
                X[i][j]=LEE.nextInt();
            }
        }
        
    }
    public static void MuestraM(int N, int M, int X[][]){
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                System.out.print(X[i][j]+" |");
            }
            System.out.println("");//cuando cambie de fila que haga un salto de lÃ­nea
        }
    }
    public static int MayorM(int X[][],int N,int M){
        int may=X[0][0];
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (X[i][j]>may) {
                    may=X[i][j];
                    
                }
                
            }
            
        }
        return may;
        
    }

                System.out.println("El mayor es: "+mayor);
    }
    
}
